```bash
> termux-setup-storage
> apt update && apt upgrade -y -y
> pkg install git 
> apt install git -y
> apt install nodejs -y
> apt install ffmpeg -y
> apt install imagemagick -y
> git clone https://github.com/Theyanderbot/THE-YANDER-BOT
> cd THE-YANDER-BOT
> ls
> npm install 
> npm install -g npm@8.7.0
> npm update
> npm start
```
